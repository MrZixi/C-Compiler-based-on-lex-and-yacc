int fact(int n){
	int temp;
    if(n==1)
        return n;
    else{
    	temp=(n*fact(n-1));
    	return temp;
    }
}


int main()
{
    int result,times;
    times=10;
    for(int i=0;i<times;i++){
    	int m = 3;
    	if( m > 1) {
        	result=fact(m);
        }
        else
            result = 1;
	}
    return 0;
}
